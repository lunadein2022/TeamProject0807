CREATE TABLE todolist (
    task_idx number PRIMARY KEY,
    yy number NULL,
    mm number NULL,
    dd number NULL,
    todo varchar2(50),
    complete number DEFAULT 0 NULL,
    client_idx number NULL,
    CONSTRAINT fk_clkey FOREIGN KEY (client_idx) REFERENCES client(client_idx)
);
